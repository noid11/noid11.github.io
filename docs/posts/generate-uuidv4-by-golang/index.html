UUIDv4 を Go で生成する

セットアップ

go mod init generate-uuidv4-by-golang
go get github.com/google/uuid

実装例

main.go

package main

import (
	"fmt"

	"github.com/google/uuid"
)

func main() {
	uuidv4, err := uuid.NewRandom()
	if err != nil {
		fmt.Println(err)
		return
	}
	fmt.Println(uuidv4.String())
}

実行例

% go run main.go
3c85b7b7-0b11-4612-9b1d-1dab37e20c9d
